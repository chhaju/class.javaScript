Calculate Average Marks

function calculateAverage() { var markA = parseFloat(document.getElementById('markA').value); var markB = parseFloat(document.getElementById('markB').value); var markC = parseFloat(document.getElementById('markC').value); var markD = parseFloat(document.getElementById('markD').value); var markE = parseFloat(document.getElementById('markE').value); var totalMarks = markA + markB + markC + markD + markE; var average = totalMarks / 5; var result = "Average marks: " + average.toFixed(2); document.getElementById("result").innerHTML = result; }
